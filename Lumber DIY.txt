;------------------------------ LUMBER ------------------------------
lumber:
set %real #skill - 9
set %change %real - %oldskill
for %cnt 1 15
{
call %reconect
if #weight > %peso
goto guardar
gosub stealth
event sysmessage Lumberjacking Lugar %target de %MAXWPTS
set #lobjectid %machado
event macro 17 0
event macro 17 0
target
click %TREEX . %target %TREEY . %target
goto jornal
;---------------------------- JORNAL -------------------------------------
jornal:
wait %pausescan
scanjournal 1
if nao_ha_madeiras in #journal
goto contas
if immune in #journal
goto contas
if the_tree_foliage in #journal
goto contas
if reach_this in #journal
goto contas
if longe in #journal
goto contas
if mochila in #journal
goto lumber
if decide_not in #journal
wait 1s
if nenhum  in #journal
{
set %fail %fail + 1
goto lumber
}
if %fail > 10
{
set %fail 0
goto contas
}
event macro 22 0  ;Last Target
set %repetir %repetir + 1
if %repetir > 60
{
set %repetir 0
goto lumber
}
goto jornal
;------------------------------ Stealth + Hiding-----------------------
sub stealth
hiding:
if #charstatus = G
key tab
if #charstatus <> H
{
 event macro 13 21
 wait 80
 event macro 13 47
 wait 80
 goto hiding
}
}
return
;--------------------------------- Guardar ----------------------------
guardar:
call %reconect
set %guardar %guardar + 1
if %guardar = 1
{
set %target2 %target
set %target 0
goto onde_cortar
}
if %guardar > 2
{
set %guardar 0
set %target %target2
goto onde_cortar
}
acharlog:
call %reconect
finditem ZLK
if #findkind = 1
{
ignoreitem #findid 1
goto acharlog
}
if #findkind = 0
{
if %tipo = banco
gosub banco
event drag #findid
wait 10
msg $
wait 10
click %guardarx %guardary p
wait 10
goto acharlog
}
}
if #findkind = -1
goto guardar
goto acharlog

;------------------------------ ONDE CORTAR --------------------------
onde_cortar:
call %reconect
for %mv 1 10
{
move %rx . %target %ry . %target a 3s
set %charposx . %mv #charposx
set %charposy . %mv #charposy
for %mv2 1 %mv
{
if #charposx = %rx . %target
{
if #charposy = %ry . %target
{
set %parado 0
if %guardar = 1
goto guardar
goto lumber
}
}
set %mv3 %mv2 - 1
if %charposx . %mv = %charposx . %mv3
{
if %charposy . %mv = %charposy . %mv3
{
set %parado %parado + 1
}
}
if %parado >= 4
goto iniciar_desvio
}
}
goto onde_cortar

iniciar_desvio:
call %reconect
if #charposx >= %rx . %target
{
set %desviar1 home
set %desviar2 pgdn
}
if #charposx <= %rx . %target
{
set %desviar1 pgdn
set %desviar2 home
}
if #charposy > %ry . %target
{
set %desviar3 end
set %desviar4 pgup
}
if #charposy < %ry . %target
{
set %desviar3 pgup
set %desviar4 end
}
set %parado 0
set %d1 x
set %d2 1
set %andou 0
set %t 1

desviar:
call %reconect
for %des 1 4
{
set %charpos . %d1 #charpos . %d1
for %des2  1 3
{
key %desviar . %d2
wait 10
if %charpos . %d1 <> #charpos . %d1
{
set %andou %andou + 1
gosub mudar_desviar . %t
}
}
gosub mudar2_desviar . %t
}
goto mudar_tipo
